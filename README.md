 function mergeArray(arr, first, mid, last, temp) {
                              var i=first;
                              var m=mid;
                              var j=mid+1;
                              var n=last;
                              var k=0;
                          while(i<=m && j<=n) {
                              if(arr[i] < arr[j]) {
                                    temp[k++] = arr[i++];
                              } else {
                                    temp[k++] = arr[j++];
                                     }
                                               }
                          while(i<=m) {
                                    temp[k++] = arr[i++];
                                      }
                          while(j<=n) {
                                    temp[k++] = arr[j++];
                                      }  
                          for(let l=0; l<k; l++) { 
                                    arr[first+l] = temp[l];
                                                }
                                     return arr;
                                 }
